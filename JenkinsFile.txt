pipeline{
    node any
    environment{
        IMAGE_TAG = 'pipeline-project'
        VERSION = 'v1.0'
        USER_ID =  'venkatsai0779'
    }
    stages{
        stage('pull repo'){
            steps{
                git branch: 'main', changelog: false, poll: false, url: 'https://github.com/venkat-0-7/jenkins-pipeline.git'
            }
        }
        stage('build code'){
            steps{
                sh 'mvn clean package'
            }
        }
        stage('build & push image'){
            steps{
                dir('jenkins-pipeline') {
                    sh 'docker build -t $USER_ID/IMAGE_TAG .'
                }
                withCredentials([usernamePassword(credentialsId: 'docker-creds', passwordVariable: 'dokcer-pass', usernameVariable: 'dokcer-user')]) {
                    sh 'docker push USER_ID/IMAGE_TAG'
                }
            }
        }
        stage('deploy container'){
            steps{
                sh 'docker run -d USER_ID/IMAGE_TAG'
            }
        }
        stage('clean up'){
            cleanWs()
        }
    }
}
